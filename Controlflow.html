C Programming – Syntax & Examples

Decision Making • Conditions • Loops

Condition

1. if Statement

Syntax

if (condition) {
    statement;
}

Example

int age = 20;

if (age >= 18) {
    printf("Eligible to vote");
}
Condition

2. if–else Statement

Syntax

if (condition) {
    statement1;
} else {
    statement2;
}

Example

if (age >= 18)
    printf("Adult");
else
    printf("Minor");
Condition

3. else–if Ladder

Syntax

if (condition1)
    statement1;
else if (condition2)
    statement2;
else
    statementN;

Example

if (marks >= 90)
    printf("Grade A");
else if (marks >= 75)
    printf("Grade B");
else
    printf("Fail");
Condition

4. Nested if

Syntax

if (condition1) {
    if (condition2) {
        statement;
    }
}

Example

if (usernameOK) {
    if (passwordOK) {
        printf("Login successful");
    }
}
Decision Making

5. switch Statement

Syntax

switch (expression) {
case value:
    statement;
    break;
default:
    statement;
}

Example

int choice = 2;

switch (choice) {
case 1:
    printf("Add");
    break;
case 2:
    printf("Subtract");
    break;
default:
    printf("Invalid");
}
Loop

6. for Loop

Syntax

for (initialization; condition; increment) {
    statement;
}

Example

for (int i = 1; i <= 5; i++) {
    printf("%d ", i);
}
Loop

7. while Loop

Syntax

while (condition) {
    statement;
}

Example

int i = 1;
while (i <= 5) {
    printf("%d ", i);
    i++;
}
Loop

8. do–while Loop

Syntax

do {
    statement;
} while (condition);

Example

int i = 1;
do {
    printf("%d ", i);
    i++;
} while (i <= 5);
Control

9. break & continue

Syntax

break;
continue;

Example

for (int i = 1; i <= 5; i++) {
    if (i == 3)
        continue;
    printf("%d ", i);
}
