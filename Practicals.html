6. Reverse a Number

Reverses the digits of an integer.

#include <stdio.h>

int main() {
    int num, rev = 0, rem;
    printf("Enter a number: ");
    scanf("%d", &num);
    while(num != 0) {
        rem = num % 10;
        rev = rev * 10 + rem;
        num /= 10;
    }
    printf("Reversed number = %d\n", rev);
    return 0;
}
Output (example):
Enter a number: 1234
Reversed number = 4321

    int num, i, flag = 0;
    printf("Enter a number: ");

10. Palindrome Number

Checks if a number is a palindrome.

#include <stdio.h>

int main() {
    int num, rev=0, rem, original;
    printf("Enter a number: ");
    scanf("%d",&num);
    original=num;
    while(num!=0){
        rem=num%10;
        rev=rev*10+rem;
        num/=10;
    }
    if(original==rev)
        printf("%d is a palindrome\n",original);
    else
        printf("%d is not a palindrome\n",original);
    return 0;
}
Output (example):
Enter a number: 121
121 is a palindrome

11. GCD of Two Numbers

Calculates the Greatest Common Divisor using Euclidean algorithm.

#include <stdio.h>

int main() {
    int a, b, temp;
    printf("Enter two numbers: ");
    scanf("%d %d", &a, &b);
    while(b != 0) {
        temp = b;
        b = a % b;
        a = temp;
    }
    printf("GCD = %d\n", a);
    return 0;
}
Output (example):
Enter two numbers: 24 36
GCD = 12

12. LCM of Two Numbers

Calculates the Least Common Multiple of two numbers.

#include <stdio.h>

int main() {
    int a, b, max;
    printf("Enter two numbers: ");
    scanf("%d %d", &a, &b);
    max = (a > b) ? a : b;
    while(1) {
        if(max % a == 0 && max % b == 0) {
            printf("LCM = %d\n", max);
            break;
        }
        max++;
    }
    return 0;
}
Output (example):
Enter two numbers: 4 6
LCM = 12

15. Bubble Sort

Sorts an array of integers in ascending order.

#include <stdio.h>

int main() {
    int n, i, j, temp;
    printf("Enter number of elements: ");
    scanf("%d", &n);
    int arr[n];
    printf("Enter elements: ");
    for(i = 0; i < n; i++)
        scanf("%d", &arr[i]);
    for(i = 0; i < n-1; i++) {
        for(j = 0; j < n-i-1; j++) {
            if(arr[j] > arr[j+1]) {
                temp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = temp;
            }
        }
    }
    printf("Sorted array: ");
    for(i = 0; i < n; i++)
        printf("%d ", arr[i]);
    printf("\n");
    return 0;
}
Output (example):
Enter number of elements: 5
Enter elements: 5 2 9 1 3
Sorted array: 1 2 3 5 9

20. Simple Calculator

Performs addition, subtraction, multiplication, and division.

#include <stdio.h>

int main() {
    char op;
    double num1, num2;
    printf("Enter an operator (+,-,*,/): ");
    scanf(" %c", &op);
    printf("Enter two operands: ");
    scanf("%lf %lf", &num1, &num2);

    switch(op) {
        case '+': printf("Result: %.2lf\n", num1+num2); break;
        case '-': printf("Result: %.2lf\n", num1-num2); break;
        case '*': printf("Result: %.2lf\n", num1*num2); break;
        case '/': 
            if(num2 != 0)
                printf("Result: %.2lf\n", num1/num2);
            else
                printf("Error: Division by zero\n");
            break;
        default: printf("Invalid operator\n");
    }
    return 0;
}
Output (example):
Enter an operator (+,-,*,/): *
Enter two operands: 5 6
Result: 30.00
